VALID

FROM employees;

SELECT name
FROM employees
WHERE salary > 50000
  AND age < 40;

SELECT name
FROM employees
WHERE salary > (
    SELECT AVG(salary)
    FROM employees
);

SELECT department, SUM(salary)
FROM employees
GROUP BY department;

SELECT department, SUM(salary)
FROM employees
GROUP BY department
HAVING SUM(salary) > 200000;

SELECT salary * 2
FROM employees
ORDER BY salary * 2;

SELECT e.name, d.name
FROM employees e
JOIN departments d
ON e.dept_id = d.id;

SELECT d.name, SUM(e.salary)
FROM employees e
JOIN departments d
ON e.dept_id = d.id
WHERE e.salary > 30000
GROUP BY d.name
HAVING SUM(e.salary) > 100000
ORDER BY d.name
LIMIT 3;

    SELECT id, salary FROM employees
);   -- subquery returns multiple columns

FROM employees
HAVING department = 'HR';   -- HAVING without GROUP BY

FROM employees
ORDER BY a b;               -- invalid syntax

FROM employees
